<!-- ========== ACCOUNTING DASHBOARD JS ========== -->
<script>
(function(){
  // Chart.js CDN يجب أن يكون محملاً في الـ parent
  // نستخدم canvas API مباشرة إذا لم يكن Chart.js متوفراً

  let accData = null;
  let accCurrentTab = 'pay';

  function fmt(n){ return Number(n||0).toLocaleString('ar-MA'); }

  window.loadAccounting = function(){
    const month = document.getElementById('acc-month')?.value || '';
    const year  = document.getElementById('acc-year')?.value  || new Date().getFullYear();
    let url = `/api/accounting?year=${year}`;
    if(month) url += `&month=${month}`;

    // Loading state
    ['acc-balance','acc-income','acc-expense','acc-late'].forEach(id=>{
      const el = document.getElementById(id);
      if(el) el.textContent = '...';
    });

    fetch(url)
      .then(r=>r.json())
      .then(d=>{
        accData = d;
        renderKPIs(d);
        renderApartmentStatus(d);
        renderChart(d);
        renderOps(d, accCurrentTab);
        renderAlerts(d);
      })
      .catch(()=>{
        // fallback: بيانات تجريبية للمعاينة
        const demo = {
          total_income: 12000, total_expense: 6600,
          payments: [{apartment_number:'11',amount:300,month:6,year:2026,date_paid:'2026-06-01'},{apartment_number:'3',amount:300,month:6,year:2026,date_paid:'2026-06-03'}],
          expenses: [{label:'فاتورة كهرباء',amount:450,date:'2026-06-05'},{label:'صيانة مصعد',amount:800,date:'2026-05-20'}],
          monthly: {'2026-01':{income:3000,expense:900},'2026-02':{income:3200,expense:1200},'2026-03':{income:2800,expense:700},'2026-04':{income:3100,expense:1400},'2026-05':{income:2900,expense:1100},'2026-06':{income:12000,expense:6600}},
          paid_apts: 11, total_apts: 17, late_amount: 2400
        };
        accData = demo;
        renderKPIs(demo);
        renderApartmentStatus(demo);
        renderChart(demo);
        renderOps(demo, accCurrentTab);
        renderAlerts(demo);
      });
  };

  function renderKPIs(d){
    const income  = d.total_income  || 0;
    const expense = d.total_expense || 0;
    const balance = income - expense;
    const late    = d.late_amount   || 0;

    const el = (id,v)=>{ const e=document.getElementById(id); if(e) e.textContent=fmt(v); };
    el('acc-balance', balance);
    el('acc-income',  income);
    el('acc-expense', expense);
    el('acc-late',    late);

    // لون الرصيد
    const balEl = document.getElementById('acc-balance');
    if(balEl && balEl.parentElement){
      const bg = balance >= 0
        ? 'linear-gradient(135deg,#059669,#10b981)'
        : 'linear-gradient(135deg,#dc2626,#ef4444)';
      balEl.parentElement.style.background = bg;
    }
  }

  function renderApartmentStatus(d){
    const paid  = d.paid_apts  || (d.payments ? [...new Set(d.payments.map(p=>p.apartment_id||p.apartment_number))].length : 0);
    const total = d.total_apts || 17;
    const late  = total - paid;
    const pct   = total > 0 ? Math.round((paid/total)*100) : 0;

    const s = (id,v)=>{ const e=document.getElementById(id); if(e) e.textContent=v; };
    s('acc-paid-count',  paid);
    s('acc-late-count',  late);
    s('acc-total-count', total);
    s('acc-progress-pct', pct + '%');

    const bar = document.getElementById('acc-progress-bar');
    if(bar){
      bar.style.width = pct + '%';
      bar.style.background = pct >= 80
        ? 'linear-gradient(90deg,#059669,#10b981)'
        : pct >= 50
          ? 'linear-gradient(90deg,#d97706,#f59e0b)'
          : 'linear-gradient(90deg,#dc2626,#f87171)';
    }
  }

  function renderChart(d){
    const canvas = document.getElementById('acc-chart');
    if(!canvas) return;
    const ctx = canvas.getContext('2d');
    const monthly = d.monthly || {};
    const keys = Object.keys(monthly).sort().slice(-6);
    const incomes  = keys.map(k=>(monthly[k]||{}).income  || 0);
    const expenses = keys.map(k=>(monthly[k]||{}).expense || 0);
    const labels   = keys.map(k=>{ const [y,m]=k.split('-'); const mn=['','يناير','فبراير','مارس','أبريل','ماي','يونيو','يوليوز','غشت','شتنبر','أكتوبر','نونبر','دجنبر']; return mn[+m]||m; });

    // رسم بياني يدوي بدون مكتبة
    const W = canvas.offsetWidth || 300;
    const H = 160;
    canvas.width  = W * (window.devicePixelRatio||1);
    canvas.height = H * (window.devicePixelRatio||1);
    ctx.scale(window.devicePixelRatio||1, window.devicePixelRatio||1);

    ctx.clearRect(0,0,W,H);
    const maxVal = Math.max(...incomes, ...expenses, 1);
    const padL=10, padR=10, padT=10, padB=30;
    const chartW = W-padL-padR;
    const chartH = H-padT-padB;
    const n = keys.length || 1;
    const barW = (chartW/n)*0.32;
    const gap  = (chartW/n)*0.36;

    // خطوط خلفية
    ctx.strokeStyle='#f1f5f9'; ctx.lineWidth=1;
    for(let i=0;i<=4;i++){
      const y = padT + (chartH/4)*i;
      ctx.beginPath(); ctx.moveTo(padL,y); ctx.lineTo(W-padR,y); ctx.stroke();
    }

    keys.forEach((k,i)=>{
      const x = padL + (chartW/n)*i + gap/2;
      // عمود المداخيل
      const ih = (incomes[i]/maxVal)*chartH;
      ctx.fillStyle='#3b82f6';
      ctx.beginPath();
      ctx.roundRect(x, padT+chartH-ih, barW, ih, [4,4,0,0]);
      ctx.fill();
      // عمود المصاريف
      const eh = (expenses[i]/maxVal)*chartH;
      ctx.fillStyle='#f87171';
      ctx.beginPath();
      ctx.roundRect(x+barW+2, padT+chartH-eh, barW, eh, [4,4,0,0]);
      ctx.fill();
      // label
      ctx.fillStyle='#64748b'; ctx.font='9px Cairo,sans-serif';
      ctx.textAlign='center';
      ctx.fillText(labels[i], x+barW+1, H-padB+14);
    });
  }

  function renderOps(d, tab){
    const el = document.getElementById('acc-ops-list');
    if(!el) return;
    let items = [];
    if(tab==='pay'){
      items = (d.payments||[]).slice(-8).reverse().map(p=>({
        icon:'💰', label: `شقة ${p.apartment_number||p.apartment_id} — شهر ${p.month}/${p.year}`,
        amount: `+${fmt(p.amount)}`, color:'#059669'
      }));
    } else {
      items = (d.expenses||[]).slice(-8).reverse().map(e=>({
        icon:'🧾', label: e.label||e.description||'مصروف',
        amount: `-${fmt(e.amount)}`, color:'#dc2626'
      }));
    }
    if(!items.length){
      el.innerHTML='<div class="empty"><div class="empty-ico">📭</div><p>لا توجد عمليات</p></div>';
      return;
    }
    el.innerHTML = items.map(it=>`
      <div style="display:flex;justify-content:space-between;align-items:center;padding:8px 10px;background:#f8fafc;border-radius:10px;border-right:3px solid ${it.color}">
        <div style="display:flex;align-items:center;gap:8px">
          <span style="font-size:16px">${it.icon}</span>
          <span style="font-size:12px;font-family:Cairo,sans-serif;font-weight:600">${it.label}</span>
        </div>
        <span style="font-size:13px;font-weight:900;color:${it.color};font-family:Cairo,sans-serif">${it.amount}</span>
      </div>`).join('');
  }

  function renderAlerts(d){
    const el = document.getElementById('acc-alerts');
    if(!el) return;
    const alerts = [];
    const income  = d.total_income  || 0;
    const expense = d.total_expense || 0;
    const paid    = d.paid_apts  || 0;
    const total   = d.total_apts || 0;
    const late    = total - paid;

    if(late > 0)
      alerts.push({ ico:'⚠️', txt: `${late} شقق لم تدفع بعد هذا الشهر`, color:'#d97706', bg:'#fffbeb' });
    if(expense > income && income > 0)
      alerts.push({ ico:'🔴', txt: 'المصاريف تتجاوز المداخيل — يُنصح بمراجعة الميزانية', color:'#dc2626', bg:'#fef2f2' });
    if(income - expense < 500 && income > 0)
      alerts.push({ ico:'🟡', txt: 'الرصيد منخفض — تأكد من تحصيل المتأخرات', color:'#d97706', bg:'#fffbeb' });
    if(late === 0 && total > 0)
      alerts.push({ ico:'✅', txt: 'ممتاز! جميع الشقق أدّت هذا الشهر', color:'#059669', bg:'#f0fdf4' });
    if(!alerts.length){ el.innerHTML=''; return; }

    el.innerHTML = alerts.map(a=>`
      <div style="display:flex;align-items:center;gap:8px;padding:9px 12px;background:${a.bg};border-radius:10px;border-right:3px solid ${a.color};margin-bottom:6px">
        <span style="font-size:16px">${a.ico}</span>
        <span style="font-size:12px;font-family:Cairo,sans-serif;font-weight:600;color:${a.color}">${a.txt}</span>
      </div>`).join('');
  }

  window.accTab = function(tab){
    accCurrentTab = tab;
    const tabs = {pay:'acc-tab-pay', exp:'acc-tab-exp'};
    Object.entries(tabs).forEach(([k,id])=>{
      const btn = document.getElementById(id);
      if(!btn) return;
      if(k===tab){
        btn.style.background='var(--p)'; btn.style.color='#fff'; btn.style.borderColor='var(--p)';
      } else {
        btn.style.background='#fff'; btn.style.color='var(--muted)'; btn.style.borderColor='var(--brd)';
      }
    });
    if(accData) renderOps(accData, tab);
  };

  // تحميل عند الانتقال لشاشة المحاسبة
  const _origNav = window.nav;
  if(typeof window.nav === 'function'){
    window.nav = function(screen){
      _origNav(screen);
      if(screen === 'accounting') setTimeout(window.loadAccounting, 100);
    };
  } else {
    // fallback: نراقب ظهور الشاشة
    const observer = new MutationObserver(()=>{
      const scr = document.getElementById('scr-accounting');
      if(scr && scr.classList.contains('on') && !scr.dataset.loaded){
        scr.dataset.loaded = '1';
        window.loadAccounting();
      }
    });
    observer.observe(document.body, {attributes:true, subtree:true, attributeFilter:['class']});
  }
})();
</script>
